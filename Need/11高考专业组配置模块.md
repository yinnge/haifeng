> 把字符串查字典表返回对应的等级

```
-- ============================================================
-- 根据选科要求字符串，查字典返回严格等级
-- 找不到时返回 0（当作不限）
-- ============================================================
CREATE OR REPLACE FUNCTION fn_get_requirement_level(p_req VARCHAR)
RETURNS SMALLINT AS $$
DECLARE
    v_level SMALLINT;
BEGIN
    IF p_req IS NULL OR p_req = '' THEN
        RETURN 0;
    END IF;

    SELECT requirement_level INTO v_level
    FROM t_subject_req_dict
    WHERE code = p_req;

    RETURN COALESCE(v_level, 0);
END;
$$ LANGUAGE plpgsql STABLE;
```

### 2.4 函数一：自动设置 `requirement_level`


```
-- ============================================================
-- 函数：写入明细时自动设置选科等级
-- ============================================================
CREATE OR REPLACE FUNCTION fn_auto_set_req_level()
RETURNS TRIGGER AS $$
BEGIN
    IF NEW.subject_requirements IS NULL OR NEW.subject_requirements = '' THEN
        NEW.requirement_level := 0;
        RETURN NEW;
    END IF;

    SELECT COALESCE(requirement_level, 0)
    INTO   NEW.requirement_level
    FROM   t_subject_req_dict
    WHERE  code = NEW.subject_requirements;

    -- 字典里找不到，默认0
    IF NOT FOUND THEN
        NEW.requirement_level := 0;
    END IF;

    RETURN NEW;
END;
$$ LANGUAGE plpgsql;

DROP TRIGGER IF EXISTS trg_ams_req_level ON t_admission_major_score;
CREATE TRIGGER trg_ams_req_level
    BEFORE INSERT OR UPDATE OF subject_requirements
    ON t_admission_major_score
    FOR EACH ROW
    EXECUTE FUNCTION fn_auto_set_req_level();

### 2.5 函数二：重算专业组聚合（分数 + 数量 + 选科要求）


```
-- ============================================================
-- 函数：重算某专业组的聚合数据
-- 触发时机：t_admission_major_score 增删改时
--
-- 聚合逻辑：
--   分数：所有行（类别行 + 专业行）一起参与聚合
--         取 MIN/MAX/AVG 时不区分类型
--   数量：
--     major_count   = major_id IS NOT NULL 的行数（具体专业）
--     category_count = major_id IS NULL 的行数（类别行）
--   选科要求：取最宽松（requirement_level 最小的那个 code）
-- ============================================================

CREATE OR REPLACE FUNCTION fn_recalc_group(p_group_id INTEGER)
RETURNS VOID AS $$
DECLARE
    v_least_req   VARCHAR(200);
    v_least_level SMALLINT;
BEGIN
    IF p_group_id IS NULL THEN RETURN; END IF;

    -- ===== 1. 聚合分数 + 数量 =====
    UPDATE t_admission_group g
    SET
        -- 录取人数：只统计专业行（类别行是聚合，不重复计）
        admission_count = (
            SELECT COALESCE(SUM(ams.admission_count), 0)
            FROM   t_admission_major_score ams
            WHERE  ams.group_id = p_group_id
            AND    ams.major_id IS NOT NULL      -- 只统计专业行
        ),

        -- 最低分（所有行里最小的）
        min_score = (
            SELECT MIN(ams.min_score)
            FROM   t_admission_major_score ams
            WHERE  ams.group_id = p_group_id
            AND    ams.min_score IS NOT NULL
        ),

        -- 最低位次（位次数值越大 = 分数越低）
        min_rank = (
            SELECT MAX(ams.min_rank)
            FROM   t_admission_major_score ams
            WHERE  ams.group_id = p_group_id
            AND    ams.min_rank IS NOT NULL
        ),

        -- 加权平均分（按录取人数加权，只用专业行）
        avg_score = (
            SELECT ROUND(
                SUM(ams.avg_score * ams.admission_count)
                / NULLIF(SUM(ams.admission_count), 0)
            , 2)
            FROM   t_admission_major_score ams
            WHERE  ams.group_id = p_group_id
            AND    ams.major_id IS NOT NULL
            AND    ams.avg_score IS NOT NULL
            AND    ams.admission_count IS NOT NULL
        ),

        -- 加权平均位次
        avg_rank = (
            SELECT ROUND(
                SUM(ams.avg_rank::NUMERIC * ams.admission_count)
                / NULLIF(SUM(ams.admission_count), 0)
            )::INTEGER
            FROM   t_admission_major_score ams
            WHERE  ams.group_id = p_group_id
            AND    ams.major_id IS NOT NULL
            AND    ams.avg_rank IS NOT NULL
            AND    ams.admission_count IS NOT NULL
        ),

        -- 最高分
        max_score = (
            SELECT MAX(ams.max_score)
            FROM   t_admission_major_score ams
            WHERE  ams.group_id = p_group_id
            AND    ams.max_score IS NOT NULL
        ),

        -- 最高位次（位次数值越小 = 分越高）
        max_rank = (
            SELECT MIN(ams.max_rank)
            FROM   t_admission_major_score ams
            WHERE  ams.group_id = p_group_id
            AND    ams.max_rank IS NOT NULL
        ),

        -- 专业数量（major_id 不为空的行）
        major_count = (
            SELECT COUNT(*)
            FROM   t_admission_major_score ams
            WHERE  ams.group_id  = p_group_id
            AND    ams.major_id  IS NOT NULL
        ),

        -- 类别数量（major_id 为空的行）
        category_count = (
            SELECT COUNT(*)
            FROM   t_admission_major_score ams
            WHERE  ams.group_id = p_group_id
            AND    ams.major_id IS NULL
        ),

        updated_at = NOW()

    WHERE g.id = p_group_id;


    -- ===== 2. 推导选科要求（取最宽松：requirement_level 最小的） =====
    SELECT ams.subject_requirements,
           ams.requirement_level
    INTO   v_least_req, v_least_level
    FROM   t_admission_major_score ams
    WHERE  ams.group_id = p_group_id
    AND    ams.subject_requirements IS NOT NULL
    AND    ams.subject_requirements <> ''
    ORDER BY ams.requirement_level ASC,
             ams.subject_requirements ASC    -- 同级别稳定排序
    LIMIT 1;

    -- 有结果则更新，没有结果则设为"不限"
    IF FOUND THEN
        UPDATE t_admission_group
        SET    subject_requirements = v_least_req,
               requirement_level   = v_least_level
        WHERE  id = p_group_id;
    ELSE
        UPDATE t_admission_group
        SET    subject_requirements = '不限',
               requirement_level   = 0
        WHERE  id = p_group_id
        AND    (subject_requirements IS NULL OR subject_requirements = '');
    END IF;

END;
$$ LANGUAGE plpgsql;
```

### 2.6 函数三：触发器函数（统一入口）

```
-- ============================================================
-- 触发器函数：明细表变更时自动重算专业组
-- ============================================================
CREATE OR REPLACE FUNCTION fn_on_major_score_changed()
RETURNS TRIGGER AS $$
DECLARE
    v_old_group_id INTEGER;
    v_new_group_id INTEGER;
BEGIN
    -- 确定新旧 group_id
    IF TG_OP = 'DELETE' THEN
        v_old_group_id := OLD.group_id;
        v_new_group_id := NULL;
    ELSIF TG_OP = 'INSERT' THEN
        v_old_group_id := NULL;
        v_new_group_id := NEW.group_id;
    ELSE  -- UPDATE
        v_old_group_id := OLD.group_id;
        v_new_group_id := NEW.group_id;
    END IF;

    -- 刷新新的 group
    IF v_new_group_id IS NOT NULL THEN
        PERFORM fn_recalc_group(v_new_group_id);
    END IF;

    -- 如果 group_id 变化了，旧的也要刷新
    IF v_old_group_id IS NOT NULL
       AND v_old_group_id IS DISTINCT FROM v_new_group_id THEN
        PERFORM fn_recalc_group(v_old_group_id);
    END IF;

    IF TG_OP = 'DELETE' THEN
        RETURN OLD;
    ELSE
        RETURN NEW;
    END IF;
END;
$$ LANGUAGE plpgsql;


-- 绑定触发器（AFTER，变更后再算）
DROP TRIGGER IF EXISTS trg_major_score_changed ON t_admission_major_score;
CREATE TRIGGER trg_major_score_changed
    AFTER INSERT OR UPDATE OR DELETE
    ON t_admission_major_score
    FOR EACH ROW
    EXECUTE FUNCTION fn_on_major_score_changed();


-- updated_at 自动更新
DROP TRIGGER IF EXISTS trg_ams_updated_at ON t_admission_major_score;
CREATE TRIGGER trg_ams_updated_at
    BEFORE UPDATE ON t_admission_major_score
    FOR EACH ROW
    EXECUTE FUNCTION fn_update_timestamp();
```

### XLSX注意事项：

1. 一个是主表，其他的都是外键，有一对多的，有1对1的，1对1的不能使得外键或者能代替外键的重复，
2. 有外键的xlsx都不会传外键id，而是能代替外键唯一的字段，比如大学名，城市名，需要联表找出id并传入
3. 关于如何识别哪一列跟数据库相对的问题，我认为处理方案是按名称，使用 EasyExcel 的 @ExcelProperty("院校名称") 注解即可，针对这次的名称，结合注释等信息自己决定就可以，后续列出来即可。
4. text[]字段的值必须用逗号格式，支持中英文逗号分隔，意思是逗号分隔，代表是不同的值
````
可以写一个自定义的 Converter：

// 伪代码示例

String val = cellData.getStringValue();
if (val == null) return null;
// 用正则同时兼容中英文逗号

````
5. 可能涉及到类型转换的问题：基础类型（如 Excel 里的 "2026-05-01" 转 Date，"1000" 转 Integer），EasyExcel 和 Spring Boot 默认就能完美转换。
6. 上传过程中遇到了错误，报第几行的具体错误，比如类型错误呀什么的
7. 使用事务保证一致性
8. 如果内容遇到了空值就默认是null，让数据库判断去吧。意思就是除了数据库指定不能为空之外，其他都可以为空
9. 正确顺序：传入的数据表，根据接口不同，有的是主表，有的不是主表
- 主表：检查表头规范性（比如检查看是否有超过规定的表头），检查unique字段的这一列是否有重复的,正确的话就传入数据
- 外键表：先检查院校名（1对1的不能重复，外键的必须在主表里面有对应的）不能重复检查外键表是否有数据，没有数据就报错，有数据就继续
- 错误处理：明确说清楚是什么报错了，比如哪一行，哪个表头，要明确信息
- 最佳实践：这就需要用到 Spring 的事务一致性。
````
