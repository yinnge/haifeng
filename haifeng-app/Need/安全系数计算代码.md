     * 查询历史录取数据（近5年，同省份+科类+批次）
     * 来源：t_admission_group
     */
    public List<HistoricalRecord> queryHistory(
            int universityId,
            String groupCode,
            String province,
            String subjectType,
            String batch,
            int planYear) {

        String sql = """
            SELECT
                ag.year,
                ag.admission_count      AS plan_count,
                ag.min_score,
                ag.min_rank,
                ag.avg_score,
                ag.avg_rank,
                ag.max_score,
                ag.max_rank,
                ag.subject_requirements,
                ag.requirement_level,
                ag.constraints,
                -- 历史线差 = 该年录取最低分 - 该年省控线
                ag.min_score - bsl.score_line  AS historical_line_diff,
                -- 是否新高考（用科类判断）
                CASE WHEN ag.subject_type IN ('物理类','历史类')
                     THEN TRUE ELSE FALSE END  AS is_new_gaokao
            FROM t_admission_group ag
            LEFT JOIN t_batch_score_line bsl
                ON bsl.province    = ag.province
               AND bsl.year        = ag.year
               AND bsl.subject_type = ag.subject_type
               AND bsl.batch       = ag.batch
            WHERE ag.university_id  = ?
              AND ag.group_code     = ?
              AND ag.province       = ?
              AND ag.subject_type   = ?
              AND ag.batch          = ?
              AND ag.year           BETWEEN ? AND ?
              AND ag.is_deleted     = FALSE
            ORDER BY ag.year DESC
            """;

        int fromYear = planYear - 6; // 最多取近5年（planYear-1 到 planYear-5）
        int toYear   = planYear - 1;

        return jdbc.query(sql,
            (rs, i) -> HistoricalRecord.builder()
                .year(rs.getInt("year"))
                .planCount(rs.getObject("plan_count", Integer.class))
                .minScore(rs.getObject("min_score", Integer.class))
                .minRank(rs.getObject("min_rank", Integer.class))
                .avgScore(rs.getObject("avg_score") != null
                    ? rs.getBigDecimal("avg_score").doubleValue() : null)
                .avgRank(rs.getObject("avg_rank", Integer.class))
                .maxScore(rs.getObject("max_score", Integer.class))
                .maxRank(rs.getObject("max_rank", Integer.class))
                .historicalLineDiff(rs.getObject("historical_line_diff", Integer.class))
                .isNewGaokao(rs.getBoolean("is_new_gaokao"))
                .subjectRequirements(rs.getString("subject_requirements"))
                .requirementLevel(rs.getShort("requirement_level"))
                .build(),
            universityId, groupCode, province, subjectType,
            batch, fromYear, toYear);
    }

     * 来源：t_score_rank

     */
    public Double queryDensity(
            int userScore,
            String province,
            String subjectType,
            int year) {

        String sql = """
            SELECT same_score_count, cumulative_count
            FROM t_score_rank
            WHERE province     = ?
              AND year         = ?
              AND subject_type = ?
              AND score        = ?
            LIMIT 1
            """;

        try {
            return jdbc.queryForObject(sql, (rs, i) -> {
                Integer same = rs.getObject("same_score_count", Integer.class);
                Integer cumul = rs.getObject("cumulative_count", Integer.class);
                if (same == null || cumul == null || cumul == 0) return null;
                // density = 同分人数 / 累计人数，值越大说明这个分段越拥挤
                return (double) same / cumul;
            }, province, year, subjectType, userScore);
        } catch (EmptyResultDataAccessException e) {
            return null;
        }
    }

@Component
public class FeatureBuilder {

    private static final double[] YEAR_WEIGHTS = {1.0, 0.8, 0.6, 0.4, 0.2};
    private static final int    LINE_DIFF_MIN  = 10;
    private static final double OUTLIER_RATIO  = 2.5;

    public SafetyFeatures build(SafetyContext ctx) {

        List<HistoricalRecord> history = ctx.getHistory();
        int baseYear = ctx.getPlanYear() - 1;

        // ① 预算历史均值（用于去极值）
        double rawAvgLine = history.stream()
            .filter(r -> r.getHistoricalLineDiff() != null
                      && r.getHistoricalLineDiff() >= LINE_DIFF_MIN)
            .mapToInt(HistoricalRecord::getHistoricalLineDiff)
            .average().orElse(0.0);

        double weightedLine = 0, weightedRank = 0;
        double totalLineW   = 0, totalRankW   = 0;
        int    validYears   = 0;

        for (HistoricalRecord rec : history) {
            int yearAgo = baseYear - rec.getYear() + 1;
            if (yearAgo < 1 || yearAgo > 5) continue;
            double w = YEAR_WEIGHTS[yearAgo - 1];
            validYears++;

            // 线差聚合
            Integer ld = rec.getHistoricalLineDiff();
            if (ld != null && ld >= LINE_DIFF_MIN) {
                if (rawAvgLine <= 0 || ld <= rawAvgLine * OUTLIER_RATIO) {
                    weightedLine += ld * w;
                    totalLineW   += w;
                }
            }

            // 位次聚合（优先avgRank，降级minRank*0.85）
            double effRank = 0;
            if (rec.getAvgRank() != null && rec.getAvgRank() > 0) {
                effRank = rec.getAvgRank();
            } else if (rec.getMinRank() != null && rec.getMinRank() > 0) {
                effRank = rec.getMinRank() * 0.85;
            }
            if (effRank > 0) {
                weightedRank += effRank * w;
                totalRankW   += w;
            }
        }

        // ② 计算 ratio
        Double lineDiffRatio = null;
        Integer userLineDiff = ctx.getUser().getUserLineDiff();
        if (userLineDiff != null && totalLineW > 0) {
            double avgHistLine = weightedLine / totalLineW;
            if (avgHistLine > 0) {
                lineDiffRatio = Math.min(userLineDiff / avgHistLine, 3.0);
            }
        }

        Double rankRatio = null;
        Integer userRank = ctx.getUser().getUserRank();
        if (userRank != null && userRank > 0 && totalRankW > 0) {
            double avgHistRank = weightedRank / totalRankW;
            rankRatio = Math.min(avgHistRank / userRank, 3.0);
        }

        // ③ 波动率（变异系数CV）
        double volatility = calcVolatility(history, baseYear);

        // ④ 招生计划比（近3年均值）
        Double planRatio = calcPlanRatio(ctx, baseYear);

        // ⑤ 数据质量
        double qualityMod = calcQuality(validYears, volatility,
            ctx.getUser().getIsNewGaokaoProvince(), history, baseYear);

        // ⑥ 选科变化说明
        String subjectNote = calcSubjectReqChange(ctx, history, baseYear);

        return SafetyFeatures.builder()
            .lineDiffRatio(lineDiffRatio)
            .rankRatio(rankRatio)
            .volatility(volatility)
            .planRatio(planRatio)
            .density(ctx.getCurrentDensity())
            .yearCount(validYears)
            .qualityMod(qualityMod)
            .isNewGaokao(ctx.getUser().getIsNewGaokaoProvince())
            .subjectReqChangeNote(subjectNote)
            .build();
    }

    // ─────────────────────────────────────
    // 波动率：变异系数 CV = stddev / mean
    // ─────────────────────────────────────
    private double calcVolatility(List<HistoricalRecord> history, int baseYear) {
        List<Integer> lines = new ArrayList<>();
        for (HistoricalRecord rec : history) {
            int ago = baseYear - rec.getYear() + 1;
            if (ago >= 1 && ago <= 5
                && rec.getHistoricalLineDiff() != null
                && rec.getHistoricalLineDiff() >= LINE_DIFF_MIN) {
                lines.add(rec.getHistoricalLineDiff());
            }
        }
        if (lines.size() < 2) return 0.5;

        double avg = lines.stream().mapToInt(i -> i).average().orElse(0);
        if (avg <= 0) return 0.5;

        double variance = lines.stream()
            .mapToDouble(d -> Math.pow(d - avg, 2))
            .average().orElse(0);
        double cv = Math.sqrt(variance) / avg;

        // 相邻大幅跳变额外惩罚
        int bigJumps = 0;
        for (int i = 0; i < lines.size() - 1; i++) {
            if (Math.abs(lines.get(i) - lines.get(i + 1)) / avg > 0.3) {
                bigJumps++;
            }
        }

        double vol = Math.min(cv / 0.35, 1.0);
        vol = Math.min(vol + bigJumps * 0.15, 1.0);
        return vol;
    }

    // ─────────────────────────────────────
    // 招生计划比（近3年均值为基准）
    // ─────────────────────────────────────
    private Double calcPlanRatio(SafetyContext ctx, int baseYear) {
        if (ctx.getCurrentYearPlan() == null
            || ctx.getCurrentYearPlan() <= 0) return null;

        double avgHist = ctx.getHistory().stream()
            .filter(r -> {
                int ago = baseYear - r.getYear() + 1;
                return ago >= 1 && ago <= 3
                    && r.getPlanCount() != null
                    && r.getPlanCount() > 0;
            })
            .mapToInt(HistoricalRecord::getPlanCount)
            .average().orElse(0.0);

        if (avgHist <= 0) return null;
        return ctx.getCurrentYearPlan() / avgHist;
    }

    // ─────────────────────────────────────
    // 数据质量修正（最低0.4）
    // ─────────────────────────────────────
    private double calcQuality(int yearCount, double volatility,
                               Boolean isNewGaokao,
                               List<HistoricalRecord> history, int baseYear) {
        double mod = 1.0;

        // 年份数量
        if      (yearCount == 1) mod *= 0.60;
        else if (yearCount == 2) mod *= 0.78;
        else if (yearCount == 3) mod *= 0.88;
        else if (yearCount == 4) mod *= 0.95;

        // 新旧高考比例
        long newCount = history.stream()
            .filter(r -> {
                int ago = baseYear - r.getYear() + 1;
                return ago >= 1 && ago <= 5
                    && Boolean.TRUE.equals(r.getIsNewGaokao());
            }).count();
        double newRatio = yearCount > 0 ? (double) newCount / yearCount : 0;

        if      (newRatio == 0)   mod *= 0.75;
        else if (newRatio < 0.5)  mod *= 0.85;
        else if (newRatio < 1.0)  mod *= 0.92;

        // 高波动
        if      (volatility > 0.7) mod *= 0.80;
        else if (volatility > 0.5) mod *= 0.90;

        // 新高考首届（全是旧数据）
        if (Boolean.TRUE.equals(isNewGaokao) && newCount == 0) {
            mod *= 0.70;
        }

        return Math.min(Math.max(mod, 0.40), 1.0);
    }

    // ─────────────────────────────────────
    // 选科要求变化说明
    // ─────────────────────────────────────
    private String calcSubjectReqChange(SafetyContext ctx,
                                        List<HistoricalRecord> history,
                                        int baseYear) {
        String current = ctx.getCurrentSubjectReq();
        if (current == null) return null;

        // 取最近一年的选科要求
        HistoricalRecord latest = history.stream()
            .filter(r -> baseYear - r.getYear() + 1 == 1)
            .findFirst().orElse(null);
        if (latest == null || latest.getSubjectRequirements() == null) return null;

        String prev = latest.getSubjectRequirements();
        if (current.equals(prev)) return null;

        // 用requirementLevel判断是收紧还是放宽
        Short curLevel = null;
        for (HistoricalRecord r : history) {
            if (baseYear - r.getYear() + 1 == 0) {
                curLevel = r.getRequirementLevel();
                break;
            }
        }
        Short prevLevel = latest.getRequirementLevel();

        if (curLevel != null && prevLevel != null) {
            if (curLevel > prevLevel) {
                return String.format("选科要求由「%s」变更为「%s」，门槛提高，竞争者减少，对已符合条件的考生有利。",
                    prev, current);
            } else if (curLevel < prevLevel) {
                return String.format("选科要求由「%s」变更为「%s」，门槛降低，竞争者增多，录取难度上升。",
                    prev, current);
            }
        }
        return String.format("选科要求由「%s」变更为「%s」，历史数据参考性下降。", prev, current);
    }
}

@Component
public class BaseScoreCalculator {

    private static final double DEFAULT_LINE_STEEPNESS = 2.8;
    private static final double DEFAULT_RANK_STEEPNESS = 2.4;

    public double calculate(SafetyFeatures f, ProvinceConfig config) {

        double lineSteep = get(config, ProvinceConfig::getLineSteepness,
                               DEFAULT_LINE_STEEPNESS);
        double rankSteep = get(config, ProvinceConfig::getRankSteepness,
                               DEFAULT_RANK_STEEPNESS);

        Double lineScore = f.getLineDiffRatio() != null
            ? asymmetric(f.getLineDiffRatio(), lineSteep) : null;
        Double rankScore = f.getRankRatio() != null
            ? asymmetric(f.getRankRatio(), rankSteep) : null;

        boolean isNew = Boolean.TRUE.equals(f.getIsNewGaokao());
        double lineW = isNew ? 0.42 : 0.62;
        double rankW = isNew ? 0.50 : 0.30;

        if (lineScore != null && rankScore != null) {
            return (lineScore * lineW + rankScore * rankW) / (lineW + rankW);
        } else if (lineScore != null) {
            return lineScore;
        } else if (rankScore != null) {
            return rankScore;
        }
        // 都没有数据，中性值
        return 0.5;
    }

    /**
     * 非对称连续Sigmoid
     * ratio < 1 → 指数压缩 → 断崖效应
     * ratio ≥ 1 → 正常sigmoid → 缓坡效应
     */
    private double asymmetric(double ratio, double steepness) {
        if (ratio <= 0) return 0.01;
        double adjusted = ratio < 1.0 ? Math.pow(ratio, 2.2) : ratio;
        return 1.0 / (1.0 + Math.exp(-steepness * (adjusted - 1.0)));
    }

    private double get(ProvinceConfig cfg,
                       java.util.function.Function<ProvinceConfig, Double> fn,
                       double def) {
        if (cfg == null) return def;
        Double v = fn.apply(cfg);
        return v != null ? v : def;
    }
}

@Component
public class RiskModifierService {

    private static final double DEFAULT_K = 0.15;

    public double applyModifiers(double base,
                                 SafetyFeatures f,
                                 ProvinceConfig config) {
        double result = base;

        // 1. 波动风险 → pullToCenter
        if (f.getVolatility() != null && f.getVolatility() > 0) {
            result = pullToCenter(result, f.getVolatility() * 0.5);
        }

        // 2. 招生计划变化
        if (f.getPlanRatio() != null) {
            result = planModifier(result, f.getPlanRatio());
        }

        // 3. 同分密度
        if (f.getDensity() != null && f.getDensity() > 0) {
            double k = (config != null && config.getDensityK() != null)
                ? config.getDensityK() : DEFAULT_K;
            result = densityModifier(result, f.getDensity(), k);
        }

        // 4. 数据质量 → pullToCenter
        if (f.getQualityMod() != null && f.getQualityMod() < 1.0) {
            result = pullToCenter(result, 1.0 - f.getQualityMod());
        }

        // 5. 选科变化修正
        if (f.getSubjectReqChangeNote() != null) {
            result = subjectReqModifier(result, f.getSubjectReqChangeNote());
        }

        // 6. 极小样本惩罚
        if (f.getPlanRatio() != null
            && f.getPlanRatio() > 0
            && f.getPlanRatio() < 0.1) {
            result = pullToCenter(result, 0.8);
        }

        return result;
    }

    private double planModifier(double score, double planRatio) {
        if (planRatio <= 0)    return score;
        if (planRatio < 0.7)   return score * 0.75 + 0.25 * 0.30;
        if (planRatio < 0.9)   return score * 0.85 + 0.15 * 0.40;
        if (planRatio > 1.3)   return Math.min(score * 1.05 + 0.02, 0.99);
        return score;
    }

    private double densityModifier(double score, double density, double k) {
        double modifier = 1.0 - ((density - 0.5) * k);
        modifier = Math.max(modifier, 0.70);
        return Math.min(score * modifier, 0.99);
    }

    private double subjectReqModifier(double score, String note) {
        if (note.contains("门槛提高")) {
            return Math.min(score + 0.05, 0.99);
        }
        if (note.contains("门槛降低")) {
            return score * 0.92;
        }
        return score * 0.96;
    }

    private double pullToCenter(double score, double strength) {
        strength = Math.min(strength, 1.0);
        return score * (1 - strength) + 0.5 * strength;
    }
}

```
@Component
public class ConstraintChecker {

    /**
     * 检查约束
     * 返回：命中的约束列表（含severity）
     */
    public ConstraintCheckResult check(
            List<ConstraintRule> rules,
            SafetyCalcUserInput user) {

        List<String> hardHit = new ArrayList<>();
        List<String> softHit = new ArrayList<>();

        for (ConstraintRule rule : rules) {
            if (!matchesExtra(rule, user)) continue;
            if (matchesMain(rule, user)) {
                if ("HARD".equals(rule.getSeverity())) {
                    hardHit.add(rule.getName());
                } else {
                    softHit.add(rule.getName());
                }
            }
        }

        return ConstraintCheckResult.builder()
            .hasHard(!hardHit.isEmpty())
            .hardReasons(hardHit)
            .softReasons(softHit)
            .build();
    }

    private boolean matchesMain(ConstraintRule rule, SafetyCalcUserInput user) {
        Object val = getField(user, rule.getCheckField());
        return evaluate(val, rule.getCheckOperator(), rule.getCheckValue());
    }

    private boolean matchesExtra(ConstraintRule rule, SafetyCalcUserInput user) {
        if (rule.getExtraField() == null) return true;
        Object val = getField(user, rule.getExtraField());
        return evaluate(val, rule.getExtraOperator(), rule.getExtraValue());
    }

    private boolean evaluate(Object fieldVal, String op, String checkVal) {
        if (fieldVal == null || op == null) return false;
        switch (op) {
            case "IS_TRUE":   return Boolean.TRUE.equals(fieldVal);
            case "IS_FALSE":  return Boolean.FALSE.equals(fieldVal);
            case "EQ":        return checkVal != null
                                  && checkVal.equals(String.valueOf(fieldVal));
            case "NEQ":       return checkVal != null
                                  && !checkVal.equals(String.valueOf(fieldVal));
            case "LT":  return toDouble(fieldVal) < toDouble(checkVal);
            case "LTE": return toDouble(fieldVal) <= toDouble(checkVal);
            case "GT":  return toDouble(fieldVal) > toDouble(checkVal);
            case "GTE": return toDouble(fieldVal) >= toDouble(checkVal);
            case "IN": {
                if (checkVal == null) return false;
                Set<String> set = Set.of(checkVal.split(","));
                return set.contains(String.valueOf(fieldVal));
            }
            case "NOT_IN": {
                if (checkVal == null) return false;
                Set<String> set = Set.of(checkVal.split(","));
                return !set.contains(String.valueOf(fieldVal));
            }
            default: return false;
        }
    }

    private Object getField(SafetyCalcUserInput user, String field) {
        if (field == null) return null;
        switch (field) {
            case "gender":          return user.getGender();
            case "height_cm":       return user.getHeightCm();
            case "is_color_blind":  return user.getIsColorBlind();
            default:                return null;
        }
    }

    private double toDouble(Object v) {
        try { return Double.parseDouble(String.valueOf(v)); }
        catch (Exception e) { return 0; }
    }

    @Data @Builder
    public static class ConstraintCheckResult {
        boolean      hasHard;
        List<String> hardReasons;
        List<String> softReasons;
    }
}

@Component
public class ExplainService {

    @Autowired
    private SafetyCalcRepository repo;

    public SafetyResult buildResult(double score,
                                    SafetyFeatures f,
                                    String version,
                                    ConstraintChecker.ConstraintCheckResult constraint) {
        // 从字典表动态读取等级
        SafetyLevelConfig level = determineLevelFromDict(score);

        // 一致性警告
        String warning = checkConsistency(f);

        // 置信度
        String confidence = calcConfidence(f, warning);

        // 建议文案
        String advice = buildAdvice(score, f, level, warning, constraint);

        return SafetyResult.builder()
            .coefficient(Math.round(score * 10000.0) / 10000.0)
            .level(level != null ? level.getLevel() : -1)
            .levelName(level != null ? level.getName() : "未知")
            .levelShort(level != null ? level.getNameShort() : "?")
            .color(level != null ? level.getColor() : "#CCCCCC")
            .levelDescription(level != null ? level.getDescription() : "")
            .confidence(confidence)
            .advice(advice)
            .blocked(false)
            .dataYearCount(f.getYearCount())
            .dataWarning(warning)
            .algoVersion(version)
            .featuresSnapshot(toJson(f))
            .build();
    }

    public SafetyResult buildBlockedResult(
            String version,
            List<String> reasons) {
        return SafetyResult.builder()
            .coefficient(0.0)
            .level(0)
            .levelName("不可报考")
            .levelShort("禁")
            .color("#999999")
            .blocked(true)
            .blockedReasons(reasons)
            .advice("存在硬性报考限制：" + String.join("、", reasons)
                + "。请排除此志愿。")
            .algoVersion(version)
            .build();
    }

    private SafetyLevelConfig determineLevelFromDict(double score) {
        return repo.queryLevelDict().stream()
            .filter(l -> score >= l.getMinCoefficient()
                      && score < l.getMaxCoefficient())
            .findFirst().orElse(null);
    }

    private String checkConsistency(SafetyFeatures f) {
        if (f.getLineDiffRatio() == null || f.getRankRatio() == null) return null;

        // 把ratio转成score再比较
        double lineSc = sigmoid(f.getLineDiffRatio(), 1.0, 2.8);
        double rankSc = sigmoid(f.getRankRatio(), 1.0, 2.4);
        double diff = Math.abs(lineSc - rankSc);

        if (diff > 0.30) {
            return lineSc > rankSc
                ? "线差法偏乐观，位次法偏悲观，可能今年整体分数偏高，建议重点参考位次数据。"
                : "位次法偏乐观，线差法偏悲观，可能今年录取线大幅调整，建议综合参考。";
        }
        if (diff > 0.15) {
            return String.format("两种方法结论有轻微差异（%.0f%%），建议综合参考。", diff * 100);
        }
        return null;
    }

    private String calcConfidence(SafetyFeatures f, String warning) {
        int score = 0;
        score += Math.min(f.getYearCount() * 10, 40);
        score += (int)(f.getQualityMod() * 30);
        if (warning == null) score += 15;
        else if (!warning.contains("建议重点")) score += 8;
        score += (int)((1.0 - (f.getVolatility() != null ? f.getVolatility() : 0.5)) * 15);
        if (f.getLineDiffRatio() != null && f.getRankRatio() != null) score += 5;

        if (score >= 75) return "HIGH";
        if (score >= 50) return "MEDIUM";
        return "LOW";
    }

    private String buildAdvice(double score, SafetyFeatures f,
                               SafetyLevelConfig level, String warning,
                               ConstraintChecker.ConstraintCheckResult constraint) {
        StringBuilder sb = new StringBuilder();

        // 核心结论
        if (level != null) {
            sb.append(String.format("综合%d年历史数据，安全系数 %.0f%%。",
                f.getYearCount(), score * 100));
            sb.append(level.getDescription());
        }

        // 缩招预警
        if (f.getPlanRatio() != null && f.getPlanRatio() < 0.8) {
            sb.append(String.format(" ⚠️ 该专业今年招生计划较近三年均值缩减 %.0f%%，竞争加剧。",
                (1 - f.getPlanRatio()) * 100));
        }

        // 密度预警
        if (f.getDensity() != null && f.getDensity() > 0.75) {
            sb.append(" ⚠️ 您处于全省同分极密集区，一分之差可能相差数百名次，请谨慎评估。");
        }

        // 波动提示
        if (f.getVolatility() != null && f.getVolatility() > 0.6) {
            sb.append(" ⚠️ 该专业历年录取分数波动较大，预测不确定性高。");
        }

        // 选科变化
        if (f.getSubjectReqChangeNote() != null) {
            sb.append(" ℹ️ ").append(f.getSubjectReqChangeNote());
        }

        // 一致性警告
        if (warning != null) {
            sb.append(" ⚠️ ").append(warning);
        }

        // 软约束提示
        if (constraint != null && !constraint.getSoftReasons().isEmpty()) {
            sb.append(" ℹ️ 注意软性限制：")
              .append(String.join("、", constraint.getSoftReasons()))
              .append("，建议提前了解。");
        }

        // 数据不足
        if (f.getYearCount() < 3) {
            sb.append(" （历史数据仅").append(f.getYearCount())
              .append("年，结果仅供参考）");
        }

        return sb.toString();
    }

    private double sigmoid(double ratio, double center, double steep) {
        return 1.0 / (1.0 + Math.exp(-steep * (ratio - center)));
    }

    private String toJson(Object obj) {
        try {
            return com.alibaba.fastjson.JSON.toJSONString(obj);
        } catch (Exception e) {
            return "{}";
        }
    }
}

## 8. 主引擎 SafetyEngine（完整版）

Java

```
@Service
public class SafetyEngine {

    private static final String ALGO_VERSION = "V1.0.0_RELEASE";

    @Autowired private FeatureBuilder       featureBuilder;
    @Autowired private BaseScoreCalculator  baseScore;
    @Autowired private RiskModifierService  riskModifier;
    @Autowired private CalibrationService   calibration;
    @Autowired private ExplainService       explain;
    @Autowired private ConstraintChecker    constraintChecker;
    @Autowired private SafetyCalcRepository repo;

    public SafetyResult calculate(SafetyContext ctx) {

        // 0. 约束检查
        Set<String> allConstraintCodes = new HashSet<>();
        if (ctx.getGroupConstraints() != null)
            allConstraintCodes.addAll(ctx.getGroupConstraints());
        if (ctx.getMajorConstraints() != null)
            allConstraintCodes.addAll(ctx.getMajorConstraints());

        List<ConstraintRule> rules = repo.queryConstraintRules(allConstraintCodes);
        ConstraintChecker.ConstraintCheckResult constraintResult =
            constraintChecker.check(rules, ctx.getUser());

        if (constraintResult.isHasHard()) {
            return explain.buildBlockedResult(ALGO_VERSION,
                constraintResult.getHardReasons());
        }

        // 1. 数据充分性检查
        if (ctx.getHistory() == null || ctx.getHistory().isEmpty()) {
            return SafetyResult.builder()
                .level(-1).levelName("数据不足")
                .levelShort("?").color("#CCCCCC")
                .confidence("LOW")
                .advice("该专业组暂无历史录取数据，无法进行预测。")
                .dataYearCount(0)
                .algoVersion(ALGO_VERSION)
                .build();
        }

        // 2. 特征构建
        SafetyFeatures features = featureBuilder.build(ctx);

        // 3. 基础评分
        double base = baseScore.calculate(features, ctx.getProvinceConfig());

        // 4. 风险修正
        double adjusted = riskModifier.applyModifiers(
            base, features, ctx.getProvinceConfig());

        // 5. 软约束小幅降分
        if (!constraintResult.getSoftReasons().isEmpty()) {
            double penalty = Math.min(
                constraintResult.getSoftReasons().size() * 0.03, 0.10);
            adjusted *= (1 - penalty);
        }

        // 6. 校准
        double calibrated = calibration.calibrate(adjusted, ctx.getProvince());

        // 7. Clamp
        calibrated = Math.min(Math.max(calibrated, 0.01), 0.99);

        // 8. 输出
        return explain.buildResult(
            calibrated, features, ALGO_VERSION, constraintResult);
    }
}
