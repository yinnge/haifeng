#### 动态算法
流程： 用户输入好了的高考档案表，然后点击查询，会先查出所有的专业明细和专业组
我们有一个安全系数表就是干这个事情的，让专业明细表的限制加上对应专业组的限制，计算限制的安全系数，此外还会计算线差，平均分，录取人数等多维度的安全系数，通过安全系数来做查询区间
【后续实现，你只需要知道流程即可】

1. 动态根据字段计算逻辑，看这个用户填写的值对应哪些constraint_code
2. 专业组与专业明细是一对多的关系，是先与专业租的动态限制作比较，看是否有吻合的，然后标记返回限制表的severity和description
3. 专业组校验完了，再校验具体的专业明细，如果有重复，那么就返回对应的severity和description
4. 限制表不经常变，可以放到redis缓存
5. 因为用户档案表的字段是固定的，对应的数据也是同一类别，可以根据具体的字段对应的特征来做动态判断
6. 难点：要注意用户选择的科目和分数的关系，你不能将subject_type，second给写死，应该让他们跟字符串【物理，化学等6个科目】动态绑定，然后再跟分数【score_subject_1等】动态绑定来实现

// ============================================================
// 约束判断引擎 - 核心类
// 从数据库读取规则，动态判断用户是否命中约束
// ============================================================

@Service
public class ConstraintEvaluator {

    // ===== 运算符枚举 =====
    private enum Operator {
        EQ,        // 等于
        NEQ,       // 不等于
        LT,        // 小于
        LTE,       // 小于等于
        GT,        // 大于
        GTE,       // 大于等于
        IS_TRUE,   // 布尔为真
        IS_FALSE,  // 布尔为假
        IN,        // 在列表中
        NOT_IN,    // 不在列表中
        OUT_RANGE  // 超出范围（如BMI）
    }

    // ===== 主方法：判断用户命中了哪些约束代码 =====
    public Set<String> getHitConstraintCodes(
            MemberGaokao gaokao,
            MemberProfile profile,
            List<ConstraintDict> allConstraints) {

        Set<String> hitCodes = new HashSet<>();

        for (ConstraintDict rule : allConstraints) {
            if (!rule.isActive()) continue;

            // 先检查附加条件（extra_field），不满足附加条件就跳过整个规则
            if (hasExtraCondition(rule)) {
                if (!evaluateExtraCondition(rule, gaokao, profile)) {
                    continue;  // 附加条件不满足，跳过（例：规则是男生<170，用户是女生）
                }
            }

            // 判断主条件
            if (evaluateMainCondition(rule, gaokao, profile)) {
                hitCodes.add(rule.getCode());
            }
        }

        return hitCodes;
    }

    // ===== 判断主条件 =====
    private boolean evaluateMainCondition(
            ConstraintDict rule,
            MemberGaokao gaokao,
            MemberProfile profile) {

        String field = rule.getCheckField();
        String op = rule.getCheckOperator();
        String value = rule.getCheckValue();

        // 获取用户字段值（gaokao 或 profile）
        Object userValue = getFieldValue(field, gaokao, profile);
        if (userValue == null) return false;  // 字段为空，不做判断

        return evaluate(userValue, Operator.valueOf(op), value, field);
    }

    // ===== 判断附加条件 =====
    private boolean evaluateExtraCondition(
            ConstraintDict rule,
            MemberGaokao gaokao,
            MemberProfile profile) {

        Object userValue = getFieldValue(rule.getExtraField(), gaokao, profile);
        if (userValue == null) return false;

        return evaluate(userValue, Operator.valueOf(rule.getExtraOperator()),
                       rule.getExtraValue(), rule.getExtraField());
    }

    // ===== 核心：执行单个条件判断 =====
    private boolean evaluate(Object userValue, Operator op, String ruleValue, String field) {
        switch (op) {
            case IS_TRUE:
                return Boolean.TRUE.equals(userValue);

            case IS_FALSE:
                return Boolean.FALSE.equals(userValue);

            case EQ:
                return userValue.toString().equals(ruleValue);

            case NEQ:
                return !userValue.toString().equals(ruleValue);

            case LT: {
                // 特殊处理 BMI（计算字段，不在 gaokao 里）
                double v = parseDouble(userValue, field);
                return v < Double.parseDouble(ruleValue);
            }

            case LTE: {
                double v = parseDouble(userValue, field);
                return v <= Double.parseDouble(ruleValue);
            }

            case GT: {
                double v = parseDouble(userValue, field);
                return v > Double.parseDouble(ruleValue);
            }

            case GTE: {
                double v = parseDouble(userValue, field);
                return v >= Double.parseDouble(ruleValue);
            }

            case IN: {
                // IN：用户值在规则列表中
                List<String> values = Arrays.asList(ruleValue.split(","));
                return values.contains(userValue.toString());
            }

            case NOT_IN: {
                List<String> values = Arrays.asList(ruleValue.split(","));
                return !values.contains(userValue.toString());
            }

            case OUT_RANGE: {
                // 格式：min,max
                String[] parts = ruleValue.split(",");
                double min = Double.parseDouble(parts[0]);
                double max = Double.parseDouble(parts[1]);
                double v = parseDouble(userValue, field);
                return v < min || v > max;
            }

            default:
                return false;
        }
    }

    // ===== 从 gaokao/profile 对象反射获取字段值 =====
    private Object getFieldValue(String field, MemberGaokao gaokao, MemberProfile profile) {
        if (field == null) return null;

        // 特殊计算字段
        if ("bmi".equals(field)) {
            return calcBMI(gaokao);
        }
        if ("gender".equals(field)) {
            return profile != null ? profile.getGender() : null;
        }

        // gaokao 字段（Java 反射）
        try {
            Field f = MemberGaokao.class.getDeclaredField(
                toCamelCase(field)  // 下划线转驼峰
            );
            f.setAccessible(true);
            return f.get(gaokao);
        } catch (Exception e) {
            return null;
        }
    }

    private Double calcBMI(MemberGaokao gaokao) {
        if (gaokao.getHeightCm() == null || gaokao.getWeightKg() == null
            || gaokao.getHeightCm() == 0) return null;
        double heightM = gaokao.getHeightCm() / 100.0;
        return gaokao.getWeightKg().doubleValue() / (heightM * heightM);
    }

    private double parseDouble(Object value, String field) {
        if (value instanceof Number) return ((Number) value).doubleValue();
        return Double.parseDouble(value.toString());
    }

    private String toCamelCase(String snakeCase) {
        StringBuilder sb = new StringBuilder();
        boolean nextUpper = false;
        for (char c : snakeCase.toCharArray()) {
            if (c == '_') { nextUpper = true; }
            else if (nextUpper) { sb.append(Character.toUpperCase(c)); nextUpper = false; }
            else { sb.append(c); }
        }
        return sb.toString();
    }

    private boolean hasExtraCondition(ConstraintDict rule) {
        return rule.getExtraField() != null && rule.getExtraOperator() != null;
    }
}

@Service
public class ConstraintMergeService {

    // 查某专业组的合并约束（专业组constraints ∪ 所有专业明细constraints）
    public Set<String> getMergedGroupConstraints(Integer groupId) {
        Set<String> merged = new HashSet<>();

        // 1. 专业组自身的静态约束
        List<String> groupConstraints = admissionGroupMapper.getConstraints(groupId);
        merged.addAll(groupConstraints);

        // 2. 该组下所有专业明细的约束（TEXT[]字段）
        List<String> majorConstraints = admissionMajorMapper.getAllConstraintsByGroupId(groupId);
        merged.addAll(majorConstraints);

        // 3. 该组下所有专业通过 t_major_constraint 关联的约束
        List<String> staticMajorConstraints = majorConstraintMapper.getConstraintsByGroupId(groupId);
        merged.addAll(staticMajorConstraints);

        return merged;  // 去重并集
    }
}
